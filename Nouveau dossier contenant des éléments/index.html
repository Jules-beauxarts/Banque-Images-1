  <script>

    const gallery = document.getElementById('gallery');

    const images = Array.from(gallery.getElementsByTagName('img'));


    gallery.addEventListener('mouseover', (e) => {

      if (e.target.tagName === 'IMG') {

        // Retirer la classe "revealed" de toutes les images

        images.forEach(img => img.classList.remove('revealed'));


        // Sélectionner 4 images aléatoires

        const shuffled = images.sort(() => 0.5 - Math.random());

        const selected = shuffled.slice(0, 4);


        // Appliquer la classe "revealed"

        selected.forEach(img => img.classList.add('revealed'));

      }

    });


    gallery.addEventListener('mouseout', () => {

      // Réinitialiser toutes les images en gris

      images.forEach(img => img.classList.remove('revealed'));

    });

  </script>
